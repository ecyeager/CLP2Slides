	\\&=\frac{16}{6}-\frac{8}{3}-\frac{4}{2}+4-\left(\frac16+\frac13-\frac12-2\right)
	\\&=2-(-2)=4
	}
\end{align*}

\sonly<2->{
\begin{align*}
\int_{-\frac\pi2}^{\frac\pi2}\Big[\,\textcolor{W1}{f(x)}-\textcolor{C2}{g(x)}\Big]\dee x
\onslide<4->{&=\int_{-\frac\pi2}^{\frac\pi2}\left(e^{\sin x} \cos x - \sin x \cos x\right)\dee x}
\sonslide<5->{
\intertext{ Let $\su{u=\sin x}$. \linebreak Then: $\sdu{\dee u = \cos x ~\dee x}$, $\su{u\left(\frac\pi2\right)=\sin\left(\frac\pi2\right)=1}$, $\su{u\left(\frac{-\pi}{2}\right)=\sin\left(\frac{-\pi}{2}\right)=-1}$.}
&=\int_{\su{-1}}^{\su{1}}\left(e^{\su{}} - \su{}\right)\sdu{}\\&=\left[e^{\su{}}-\frac12\su{u}^2\right]_{\su{-1}}^{\su{1}}
\\&=e-\frac1e}
\end{align*}}

\only<-\sl>{

\draw[C2,very thick] plot[domain=-.5:1.5,smooth](\x,{\x*exp(1-\x*\x)})node[right]{$g(x)=xe^{1-x^2}$};
\fill[W1,opacity=0.3] plot[domain=1:0,smooth](\x,{\x*exp(\x*\x)})--plot[domain=0:1,smooth](\x,{\x*exp(1-\x*\x)});
\sonslide<2->{\xcoord{0.707}{\frac{1}{\sqrt 2}}}
\end{tikzpicture}

\sonslide<2->{
Intersections at $x=0$ and $ x = \pm\frac{1}{\sqrt 2}$:
\begin{align*}
xe^{x^2}&=xe^{1-x^2}\\
e^{x^2}&=e^{1-x^2}\text{ or }x=0\\
x^2&=1-x^2\\
 x &= \pm\frac{1}{\sqrt 2}
\end{align*}
}
\end{multicols}}

\sonslide<2->{
\small
\begin{align*}
\text{Area}&=\int_0^{\frac{1}{\sqrt 2}}\Big[\textcolor{C2}{g(x)}-\textcolor{W1}{f(x)}\Big]\dee x+\int_{\frac{1}{\sqrt 2}}^1\Big[\,\textcolor{W1}{f(x)}-\textcolor{C2}{g(x)}\Big]\dee x\\
\only<3->{&=\int_0^{\frac{1}{\sqrt 2}}\Big[xe^{1-x^2}-xe^{x^2}\Big]\dee x+\int_{\frac{1}{\sqrt 2}}^1\Big[xe^{x^2}-xe^{1-x^2}\Big]\dee x \\}
\only<4->{\color{black}
\text{Aside: }&\color{black}\underbrace{\int \sdu{x}e^{\su{1-x^2}}\sdu{\dee x}}_{\su{u=1-x^2},~\sdu{\dee u = -2x\,\dee x}}=-\frac12e^{1-x^2} +C\qquad
\underbrace{\int \sdu{x}e^{\su{x^2}}\sdu{\dee x}}_{\su{u=x^2},~\sdu{\dee u = 2x\,\dee x}}=\frac12e^{x^2} +C
\\\text{Area}&=
\left[-\frac12e^{1-x^2}-\frac12e^{x^2} \right]_0^{\frac{1}{\sqrt 2}}
+
\left[\frac12e^{x^2} -\left(-\frac12e^{1-x^2}\right)\right]_{\frac{1}{\sqrt 2}}^1\\
&=-\frac12\left[\left(e^{\frac12}+e^{\frac12}\right)-\left(e^1+e^0\right)\right]
+
\frac12\left[\left(e^1+e^0\right)-\left(e^{\frac12}+e^{\frac12}\right)\right]\\
&=e-2\sqrt e+1
}
\end{align*}
}
