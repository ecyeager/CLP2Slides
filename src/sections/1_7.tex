\begin{frame}{Reverse the Product Rule}
Product Rule:
\[\diff{}{x}\big\{u(x)\cdot v(x)\big\}= u'(x)\cdot v(x) + u(x)\cdot v'(x)
\]\pause \vfill
Related fact:
\begin{align*} 
 \int\Big[  &u'(x)\cdot v(x) + u(x)\cdot v'(x)\Big] \dee x=u(x)\cdot v(x) +C
\onslide<3->{\intertext{Rearrange:}
\implies\int\Big[ & u'(x) v(x)\Big]\dee x+\int\Big[u(x) v'(x)\Big] \dee x=u(x) v(x) +C\\
\implies\color{C3}\int\Big[&\color{C3} u(x)v'(x)\Big]\dee x=u(x)v (x) - \int \Big[v(x)u'(x) \Big]\dee x+C}
\end{align*}
\end{frame}
%----------------------------------------------------------------------------------------
%----------------------------------------------------------------------------------------
%----------------------------------------------------------------------------------------
\begin{frame}[t]{Integration by Parts}
\AnswerYes<1>\QuestionBar<1>{1}{4}
\AnswerBar<2>{1}{4}
\[\int\Big[ u(x)v'(x)\Big]\dee x=u(x)v (x)- \int \Big[v(x)u'(x) \Big]\dee x\]
Example: $\ds\int \textcolor{C2}x\textcolor{W1}{e^x} \dee x $\\
\sonslide<2->{Let  $\textcolor{C2}{u(x)=x}$ and $\textcolor{W1}{v'(x)=e^x}.$ (We'll talk \textbf{later} about choosing these)\\
Then $\textcolor{C2}{u'(x)=1}$ and $\textcolor{W1}{v(x)=e^x}$.
	\begin{align*}
	\int\Big[ \textcolor{C2}{u(x)}\textcolor{W1}{v'(x)}\Big]\dee x=\textcolor{C2}{u(x)}\textcolor{W1}{v (x)}&- \int \Big[\textcolor{W1}{v(x)}\textcolor{C2}{u'(x)} \Big]\dee x\\
	\int\Big[ \textcolor{C2}{x}\textcolor{W1}{e^x}\Big]\dee x=\textcolor{C2}{x}(\textcolor{W1}{e^x})&- \int \Big[\textcolor{W1}{(e^x)}\textcolor{C2}{1} \Big]\dee x+C\\
	\int xe^x=xe^x&-\int(e^x)\dee x+C\\
	=xe^x&-e^x+C
\end{align*}
}
\end{frame}
%----------------------------------------------------------------------------------------
%----------------------------------------------------------------------------------------
\CheckFrame{\AnswerBar{1}{4}
In the previous slide, we evaluated
\[\int xe^x \dee x = xe^x-e^x+C \]
for some constant $C$. We can check that this is correct by differentiating.}{
\begin{align*}
\diff{}{x}\Big\{ xe^x-e^x+C\Big\}&=(xe^x+e^x)-e^x=xe^x
\end{align*}
\vfill
We used the \emph{product rule} to differentiate. Remember integration by parts helps us to reverse the product rule.}

\begin{frame}[t]{Integration by Parts (IBP): A Closer Look}
	\begin{align*}
	\int\Big[ \textcolor{C2}{u(x)}\textcolor{W1}{v'(x)}\Big]\dee x=\textcolor{C2}{u(x)}\textcolor{W1}{v (x)}&- \int \Big[\textcolor{W1}{v(x)}\textcolor{C2}{u'(x)} \Big]\dee x+C\\
	\underbrace{\int \textcolor{C2}{x}\textcolor{W1}{e^x}\dee x}_{\text{How to integrate??}}
	=\textcolor{C2}{x}(\textcolor{W1}{e^x})&- 
\underbrace{\textcolor{C2}{1} \int \textcolor{W1}{e^x}\dee x}_{\text{Easy to integrate!}}+C
	\end{align*}
	\vfill
 We start and end with an integral. IBP is only useful if the new integral is somehow an improvement.
 \\
\vfill
We \textcolor{C2}{differentiate} the function we choose as $\textcolor{C2}{u(x)}$, and \textcolor{W1}{antidifferentiate} the function we choose as $\textcolor{W1}{v'(x)}$

\end{frame}

\begin{align*}
\int\Big[ \textcolor{C2}{u(x)}\textcolor{W1}{v'(x)}\Big]\dee x&=\textcolor{C2}{u(x)}\textcolor{W1}{v (x)}- \int \Big[\textcolor{W1}{v(x)}\textcolor{C2}{u'(x)} \Big]\dee x+C

We abbreviate:
\begin{itemize}
	\item\color{C2}$u(x) \rightarrow u$
	\item\color{C2}$u'(x)\,\dee x \rightarrow \ \dee u$
	\item\color{W1}$v(x) \rightarrow v$
	\item\color{W1}$v'(x)\,\dee x \rightarrow  \dee v$
	\end{itemize}
\end{frame}

\end{align*}

&=e^{2x}\sin x
\end{align*}
Our answer, strange though it looks, is the correct antiderivative.}
